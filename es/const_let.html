    两个for循环中声明的变量在不同的作用域

    • 闭包问题
    const elems = [{}, {}, {}]

for (var i = 0; i < 3; i++) {
	elems[i].onClick = function () {
		console.log(i)
	}
}

elems[0].onClick() //3

// 使用闭包

const elems = [{}, {}, {}]

for (var i = 0; i < 3; i++) {
	elems[i].onClick = (function (i) {
		return function () {
			console.log(i)
		}
	})(i)
}

elems[0].onClick() //0


// 使用let
const elems = [{}, {}, {}]

for (let i = 0; i < 3; i++) {
	elems[i].onClick = function () {
		console.log(i)
	}
}

elems[0].onClick() //0
