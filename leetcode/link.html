/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */

/*
	先构建一个节点的链表，然后遍历剩余链表的节点，使用头插法更新链表
 */

//解答成功:
//执行耗时:0 ms,击败了100.00% 的Go用户
//内存消耗:2.5 MB,击败了97.43% 的Go用户
func reverseList(head *ListNode) *ListNode {
	//  空链表
	if head == nil{
		return head
	}
	// 至少一个节点
	temp :=  head.Next
	head.Next = nil

	for temp!=nil{
		cur := temp.Next
		temp.Next = head
		head =temp
		temp = cur
	}
	return head
}
